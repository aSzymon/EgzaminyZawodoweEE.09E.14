-- kw1

SELECT id, nr_rejsu, czas kierunek, status_lotu
FROM odloty
ORDER BY czas DESC;

-- kw2 

SELECT MIN(czas)
FROM odloty
WHERE czas LIKE "%10:1%";

-- kw 3

SELECT odloty.nr_rejsu, samoloty.linie
FROM odloty INNER JOIN samoloty
on odloty.samoloty_id = samoloty.id
WHERE odloty.kierunek = "Malta";

-- kw 4 

UPDATE przyloty
SET status_lotu = "planowy"
WHERE nr_rejsu = "LH9829";
