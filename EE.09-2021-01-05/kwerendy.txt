-- kw1

SELECT czas, kierunek, nr_rejsu, status_lotu
FROM przyloty
ORDER BY czas ASC;

-- kw2

SELECT COUNT(*) 
FROM przyloty 
WHERE czas LIKE "10%";

-- kw3

SELECT przyloty.nr_rejsu, samoloty.typ
FROM przyloty INNER JOIN samoloty
ON przyloty.id = samoloty.id
WHERE przyloty.kierunek = "Warszawa";

-- kw4

UPDATE odloty
SET status_lotu = "opoznienie 20 min"
WHERE nr_rejsu = "LX5673";
